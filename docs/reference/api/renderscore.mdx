## Description
Initiates a rendering of the given score. 

## Signatures

<TypeTable>
    <TypeRow type="js" name="renderScore(score: Score, trackIndexes?: number[]): void;" />
    <TypeRow type="net" name="void RenderScore(AlphaTab.Model.Score score, IList<double>? trackIndexes)" />
    <TypeRow type="android" name="fun renderScore(score: alphaTab.model.Score, trackIndexes: alphaTab.collections.List<Double>?): Unit" />
</TypeTable>

### Parameters 

<ParameterTable>
    <ParameterRow platform="js" name="score" type="alphaTab.model.Score">
        The score containing the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="net" name="score" type="AlphaTab.Model.Score">
        The score containing the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="android" name="score" type="alphaTab.model.Score">
        The score containing the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="js" name="trackIndexes" type="number[]?">
        The indexes of the tracks from the song that should be rendered. If not provided, the first track of the
        song will be shown.
    </ParameterRow>
    <ParameterRow platform="net" name="trackIndexes" type="IList<double>?">
        The indexes of the tracks from the song that should be rendered. If not provided, the first track of the
        song will be shown.
    </ParameterRow>
    <ParameterRow platform="android" name="trackIndexes" type="alphaTab.collections.List<Double>?">
        The indexes of the tracks from the song that should be rendered. If not provided, the first track of the
        song will be shown.
    </ParameterRow>
</ParameterTable>

### Returns 
Nothing

## Description
Initiates a rendering of the score. 

## Signatures

<TypeTable>
    <TypeRow type="js" name="function renderScore(score)" />
    <TypeRow type="js" name="function renderScore(score, tracks)" />
    <TypeRow type="jquery" name="alphaTab('renderScore', score, tracks)" />
    <TypeRow type="jquery" name="alphaTab('renderScore', tracks)" />
    <TypeRow type="net" name="bool RenderScore(Score score)" />
    <TypeRow type="net" name="bool RenderScore(Score score, int[] tracks)" />
</TypeTable>

### Parameters

<ParameterTable>
    <ParameterRow platform="net" name="score" type="AlphaTab.Model.Score">
        The score that contains the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="net" name="tracks" type="AlphaTab.Model.Track[]">
        The indexes of the tracks that should be rendered. If not provided, the first track of the song will be rendered.
    </ParameterRow>
    <ParameterRow platform="js" name="score" type="alphaTab.model.Score">
        The score that contains the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="js" name="tracks" type="alphaTab.model.Track[]">
        The indexes of the tracks that should be rendered. If not provided, the first track of the song will be rendered.
    </ParameterRow>    
</ParameterTable>

### Returns
Nothing

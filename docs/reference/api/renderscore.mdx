## Description
Initiates a rendering of the score. 

## Signatures

<TypeTable>
    <TypeRow type="js" name="function renderScore(score)" />
    <TypeRow type="js" name="function renderScore(score, trackIndexes)" />
    <TypeRow type="jquery" name="alphaTab('renderScore', score, trackIndexes)" />
    <TypeRow type="jquery" name="alphaTab('renderScore', trackIndexes)" />
    <TypeRow type="net" name="bool RenderScore(Score score)" />
    <TypeRow type="net" name="bool RenderScore(Score score, int[] trackIndexes)" />
    <TypeRow type="android" name="fun renderScore(score: alphaTab.model.Score, trackIndexes: alphaTab.collections.DoubleList? = null): Unit" />
</TypeTable>

### Parameters

<ParameterTable>
    <ParameterRow platform="js" name="score" type="alphaTab.model.Score">
        The score that contains the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="js" name="tracks" type="number[]">
        The indexes of the tracks that should be rendered. If not provided, the first track of the song will be rendered.
    </ParameterRow>    
    <ParameterRow platform="net" name="score" type="AlphaTab.Model.Score">
        The score that contains the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="net" name="trackIndexes" type="double[]">
        The indexes of the tracks that should be rendered. If not provided, the first track of the song will be rendered.
    </ParameterRow>
    <ParameterRow platform="android" name="trackIndexes" type="alphaTab.model.Score">
        The score that contains the tracks to be rendered.
    </ParameterRow>
    <ParameterRow platform="android" name="trackIndexes" type="alphaTab.collections.DoubleList">
        The indexes of the tracks that should be rendered. If not provided, the first track of the song will be rendered.
    </ParameterRow>
</ParameterTable>

### Returns
Nothing

## Description
Tells alphaTab to render the given alphaTex. 

## Signatures

<TypeTable>
    <TypeRow type="js" name="tex(tex: string, tracks?: number[]): void;" />
    <TypeRow type="net" name="void Tex(string tex, IList<double>? tracks)" />
    <TypeRow type="android" name="fun tex(tex: String, tracks: alphaTab.collections.List<Double>?): Unit" />
</TypeTable>

### Parameters 

<ParameterTable>
    <ParameterRow platform="js" name="tex" type="string">
        The alphaTex code to render.
    </ParameterRow>
    <ParameterRow platform="net" name="tex" type="string">
        The alphaTex code to render.
    </ParameterRow>
    <ParameterRow platform="android" name="tex" type="String">
        The alphaTex code to render.
    </ParameterRow>
    <ParameterRow platform="js" name="tracks" type="number[]?">
        If set, the given tracks will be rendered, otherwise the first track only will be rendered.
    </ParameterRow>
    <ParameterRow platform="net" name="tracks" type="IList<double>?">
        If set, the given tracks will be rendered, otherwise the first track only will be rendered.
    </ParameterRow>
    <ParameterRow platform="android" name="tracks" type="alphaTab.collections.List<Double>?">
        If set, the given tracks will be rendered, otherwise the first track only will be rendered.
    </ParameterRow>
</ParameterTable>

### Returns 
Nothing

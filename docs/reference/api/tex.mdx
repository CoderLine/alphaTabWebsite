## Description
Tells alphaTab to render the given alphaTex.

## Signatures

<TypeTable>
    <TypeRow type="js" name="function tex(tex)" />
    <TypeRow type="js" name="function tex(tex, tracks)" />
    <TypeRow type="jquery" name="alphaTab('tex', tex)" />
    <TypeRow type="jquery" name="alphaTab('tex', tex, tracks)" />
    <TypeRow type="net" name="void Tex(string tex)" />
    <TypeRow type="net" name="void Tex(string tex, int[] tracks)" />
</TypeTable>

### Parameters

<ParameterTable>
    <ParameterRow platform="all" name="tex" type="string">
        The alphaTex encoded string to load.
    </ParameterRow>
    <ParameterRow platform="all" name="tracks" type="int[]">
        Which tracks to display.
    </ParameterRow>
    <ParameterRow platform="all" name="tracks" type="int">
        Which track to display.
    </ParameterRow>
</ParameterTable>

### Returns
Nothing

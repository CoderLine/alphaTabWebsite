### Description
Tells alphaTab to render the given alphaTex.

<Tabs defaultValue="js" values={[{label: "JavaScript", value: "js"},{label: "C#", value: "cs"},{label:"Kotlin", value: "kt"}]}>
<TabItem value="js">
<CodeBlock language="ts">{`tex(tex: string, tracks?: number[]): void;`}</CodeBlock>
</TabItem>
<TabItem value="cs">
<CodeBlock language="csharp">{`void Tex(string tex, IList<double>? tracks)`}</CodeBlock>
</TabItem>
<TabItem value="kt">
<CodeBlock language="kotlin">{`fun tex(tex: String, tracks: alphaTab.collections.DoubleList): Unit`}</CodeBlock>
</TabItem>
</Tabs>


<ParameterTable>
    <ParameterRow platform="all" name="tex">
        The alphaTex code to render.
    </ParameterRow>
    <ParameterRow platform="all" name="tracks">
        If set, the given tracks will be rendered, otherwise the first track only will be rendered.
    </ParameterRow>
</ParameterTable>

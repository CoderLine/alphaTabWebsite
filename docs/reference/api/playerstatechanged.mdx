## PlayerStateChangedEventArgs Properties

<ParameterTable>
    <ParameterRow platform="js" name="state" type="alphaTab.synth.PlayerState">
        The new state of the player. <br />
        `0` - Paused
        `1` - Playing
    </ParameterRow>
    <ParameterRow platform="net" name="State" type="AlphaTab.Synth.PlayerState">
        The new state of the player.  
    </ParameterRow>
    <ParameterRow platform="net" name="state" type="alphaTab.synth.PlayerState">
        The new state of the player.
    </ParameterRow>
    <ParameterRow platform="all" name="stopped" type="bool">
        Whether the playback was fully stopped (including jump to start) or only paused.
    </ParameterRow>
</ParameterTable>

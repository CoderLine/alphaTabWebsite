import { CodeBadge } from '@site/src/components/CodeBadge';
import { MarkdownString } from '@site/src/components/MarkdownString'

export const VibratoSettings = [];

export function toPascalCase(v) {
    return v.substr(0, 1).toUpperCase() + v.substr(1);
}

export function VibratoRow({setting}) {  
    return (
        <tr>
            <td>
                <CodeBadge type="js" name={setting.name} /><br />
                <CodeBadge type="json" name={setting.name} /><br />
                <CodeBadge type="html" name={setting.name.toLowerCase()} /><br />
                <CodeBadge type="net" name={toPascalCase(setting.name)} /><br />
                <CodeBadge type="android" name={setting.name} />
            </td>
            <td>
                <CodeBadge type="all" name={setting.type} />
            </td>
            <td><MarkdownString content={setting.defaultValue} /></td>
            <td>
                {setting.description}
            </td>
        </tr>
    );
}

export function VibratoTable() {
    const settings = VibratoSettings.map(c => <VibratoRow setting={c} />);
    return (
        <table className="table table-striped table-condensed type-table">
            <thead>
                <tr>
                    <th>Setting</th>
                    <th>Type</th>
                    <th>Default</th>
                    <th>Summary</th>
                </tr>
            </thead>    
            <tbody> 
                {settings}
            </tbody>
        </table>
    );
}
